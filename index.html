function imgforbase64(str){
        var id = 'J_preview',
            img;
            if(!document.getElementById(id)){
              img = document.createElement('img');
              img.id = id;
              img.src = 'data:image/png;base64,'+str;
              document.body.appendChild(img);
            }else{
              img  = document.getElementById(id);
              img.src = 'data:image/png;base64,'+str;
            }
        console.log(str);
      }
